		\hline 
		\textbf{} & \textbf{Löwe} & \textbf{Mensch} \\ \hline  
		\textbf{Gewicht des Gehirns} & 1 & 2 \\  
		\textbf{Körpertemperatur} & 3 & 4 \\  
		\textbf{Atemfrequenz} & 5 & 6 \\ 
		\textbf{Maximale Geschwindigkeit} & 7 & 8 \\  
		\textbf{Grundstoffwechselumsatz} & 9 & 10 \\ \hline
	\end{tabular} 
	\label{}
\end{table}
		\end{center}

	\end{outputbox}
